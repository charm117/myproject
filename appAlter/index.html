单提示 带标题 带确认和取消
复制前请选阅读源码，再修改成自己的 
            let tips = new ShenTips();
        
            tips.tips({
                msg: "你今天是否更努力了呢？", 
                time: 500
            });
        
            tips.alert({
                title: "提示",
                msg: "你今天是否更努力了呢？", 
                time: 500, 
                options: ["是的呢"],
                confirm: function() { 
                    console.log("我比昨天更加努力了");
                }
            })
        
            tips.confirm({
                title: "请确认",
                msg: "是否提交",
                options: ["继续选择", "提交选择"],
                time: 500,
                close: false,
                cancel() {
                    console.log("点击了取消按钮");
                },
                confirm() {
                    console.log("点击了确认按钮");
                    that.close();
                }
            })
